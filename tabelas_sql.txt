
CREATE TABLE USER
(ID 	  INTEGER AUTO_INCREMENT,
 NAME 	  VARCHAR(255) NOT NULL,
 PASSWORD VARCHAR(255),
 CODE 	  VARCHAR(255),
 PRIMARY KEY (ID, NAME)
);

CREATE TABLE ACCOUNT
(ID 	 INTEGER AUTO_INCREMENT,
 NAME 	 VARCHAR(255) NOT NULL,
 TIPO 	 CHARACTER(1),
 BALANCE FLOAT,
 USER_ID INTEGER,
 PRIMARY KEY (ID, NAME),
 FOREIGN KEY (USER_ID) REFERENCES USER(ID)
);

CREATE TABLE RELEASE_TYPE
(ID 	 INTEGER AUTO_INCREMENT,
 NAME 	 VARCHAR(255) NOT NULL,
 USER_ID INTEGER,
 PRIMARY KEY (ID, NAME),
 FOREIGN KEY (USER_ID) REFERENCES USER(ID)
);

CREATE TABLE RELEASE
(ID 	  INTEGER AUTO_INCREMENT,
 VALUE    FLOAT,
 PAY_TYPE VARCHAR(255),
 DESCP    VARCHAR(255),
 OP	  VARCHAR(3),
 DATE	  VARCHAR(10),
 ACC_ID   INTEGER,
 REL_TYPE INTEGER,
 PRIMARY KEY (ID),
 FOREIGN KEY (ACC_ID) REFERENCES ACCOUNT(ID)
 FOREIGN KEY (REL_TYPE) REFERENCES RELEASE_TYPE(ID)
);
